**Script for creating tables**
CREATE TABLE CUSTOMER (
    Customer_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    Full_Name VARCHAR2(100) NOT NULL,
    Phone VARCHAR2(20),
    Email VARCHAR2(100),
    Registration_Date DATE DEFAULT SYSDATE
);
