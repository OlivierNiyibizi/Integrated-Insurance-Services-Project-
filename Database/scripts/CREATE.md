**Script for creating tables**

CREATE TABLE CUSTOMER (
    Customer_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    
    Full_Name VARCHAR2(100) NOT NULL,
    
    Phone VARCHAR2(20),
    
    Email VARCHAR2(100),
    
    Registration_Date DATE DEFAULT SYSDATE
);

**table of policy**

CREATE TABLE POLICY (
    Policy_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    
    Policy_name VARCHAR2(100) NOT NULL,
    
    Policy_type VARCHAR2(50) NOT NULL,
    
    Premium_amount NUMBER(10,2) NOT NULL,
    
    Coverage_amount NUMBER(10,2) NOT NULL,
    
    Duration_months NUMBER NOT NULL
);

**Table of customer_policy**

CREATE TABLE CUSTOMER_POLICY (
    Customer_policy_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
   
    Customer_id NUMBER NOT NULL,
    
    Policy_id NUMBER NOT NULL,
    
    Start_date DATE DEFAULT SYSDATE,
    
    End_date DATE,
    
    Status VARCHAR2(20) DEFAULT 'Active',
    
    CONSTRAINT fk_customer FOREIGN KEY (Customer_id) REFERENCES CUSTOMER(Customer_id),
    
    CONSTRAINT fk_policy FOREIGN KEY (Policy_id) REFERENCES POLICY(Policy_id)
);

**Table of payments**

CREATE TABLE PAYMENT (
    Payment_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    
    Customer_policy_id NUMBER NOT NULL,
    
    Payment_date DATE DEFAULT SYSDATE,
    
    Amount_paid NUMBER(10,2) NOT NULL,
    
    Payment_method VARCHAR2(50),
    
    CONSTRAINT fk_customer_policy_payment FOREIGN KEY (Customer_policy_id) REFERENCES CUSTOMER_POLICY(Customer_policy_id)
);

**Table of claim**

CREATE TABLE CLAIM (
    Claim_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    
    Customer_policy_id NUMBER NOT NULL,
    
    Claim_date DATE DEFAULT SYSDATE,
    
    Claim_description VARCHAR2(500),
    
    Claim_status VARCHAR2(20) DEFAULT 'Pending',
    
    Approved_amount NUMBER(10,2),
    
    CONSTRAINT fk_customer_policy_claim FOREIGN KEY (Customer_policy_id) REFERENCES CUSTOMER_POLICY(Customer_policy_id)
);
